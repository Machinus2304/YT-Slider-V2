<script>

let currentIndex = 0;

const videos = document.querySelectorAll('#youtube-slider iframe');


function showNextVideo() {

  videos[currentIndex].style.display = 'none';

  currentIndex = (currentIndex + 1) % videos.length;

  videos[currentIndex].style.display = 'block';

}


videos[currentIndex].style.display = 'block';

setInterval(showNextVideo, 10000); // Wechselt alle 10 Sekunden

</script>
